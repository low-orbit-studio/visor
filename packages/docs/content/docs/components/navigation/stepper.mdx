title: Stepper
description: A multi-step progress indicator with horizontal and vertical orientations. Auto-derives step status from activeStep with accessible ARIA roles.
---

import { ComponentPreview } from '@/components/preview';

  Stepper,

  StepperTitle,
  StepperDescription,
  StepperSeparator,
} from '@/components/ui/stepper';

## Installation

```bash
npx visor add stepper
```

This copies two files into your project:

- `components/ui/stepper/stepper.tsx` — the component
- `components/ui/stepper/stepper.module.css` — the styles

## Usage

```tsx
import {
  Stepper,
  StepperItem,
  StepperTrigger,
  StepperTitle,
  StepperSeparator,
} from '@/components/ui/stepper/stepper';

export default function Example() {
  return (
    <Stepper activeStep={1}>
      <StepperItem step={0}>
        <StepperTrigger step={0} />
        <StepperTitle>Account</StepperTitle>
      </StepperItem>
      <StepperSeparator complete />
      <StepperItem step={1}>
        <StepperTrigger step={1} />
        <StepperTitle>Profile</StepperTitle>
      </StepperItem>
      <StepperSeparator />
      <StepperItem step={2}>
        <StepperTrigger step={2} />
        <StepperTitle>Review</StepperTitle>
      </StepperItem>
    </Stepper>
  );
}
```

## Sub-components

| Component | Element | Purpose |
|-----------|---------|---------|
| `Stepper` | `<div>` | Root container with context provider |
| `StepperItem` | `<div>` | Step wrapper, auto-derives status |
| `StepperTrigger` | `<button>` | Clickable step indicator circle |
| `StepperTitle` | `<div>` | Step label |
| `StepperDescription` | `<p>` | Step description |
| `StepperSeparator` | `<div>` | Connecting line between steps |
