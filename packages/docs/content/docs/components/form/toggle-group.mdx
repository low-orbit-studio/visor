---
title: Toggle Group
description: A group of toggle buttons where one or multiple can be active. Built on Radix UI with CVA variants.
---

import { ComponentPreview } from '@/components/preview';
import { PropsTable } from '@/components/props-table';
import { ToggleGroup, ToggleGroupItem } from '@/components/ui/toggle-group';

## Installation

```bash
npx visor add toggle-group
```

## Usage

```tsx
import { ToggleGroup, ToggleGroupItem } from '@/components/ui/toggle-group/toggle-group';

<ToggleGroup type="single">
  <ToggleGroupItem value="a">A</ToggleGroupItem>
  <ToggleGroupItem value="b">B</ToggleGroupItem>
</ToggleGroup>
```
