## Installation

```bash
npx visor add fieldset
```

## Usage

```tsx
import { Fieldset, FieldsetLegend } from '@/components/ui/fieldset/fieldset';

<Fieldset>
  <FieldsetLegend>Contact Information</FieldsetLegend>
  {/* form fields */}
</Fieldset>
```
