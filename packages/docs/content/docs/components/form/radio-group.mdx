## Installation

```bash
npx visor add radio-group
```

## Usage

```tsx
import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group/radio-group';

<RadioGroup defaultValue="a">
  <RadioGroupItem value="a" />
  <RadioGroupItem value="b" />
</RadioGroup>
```
