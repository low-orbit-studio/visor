---

import { ComponentPreview } from '@/components/preview';
import { PropsTable } from '@/components/props-table';
import { Combobox, ComboboxInput, ComboboxContent, ComboboxItem, ComboboxEmpty } from '@/components/ui/combobox';

## Installation

```bash
npx visor add combobox
```

## Usage

```tsx
import { Combobox, ComboboxInput, ComboboxContent, ComboboxItem, ComboboxEmpty } from '@/components/ui/combobox/combobox';

<Combobox>
  <ComboboxInput placeholder="Pick a fruit..." />
  <ComboboxContent>
    <ComboboxItem value="apple">Apple</ComboboxItem>
    <ComboboxItem value="banana">Banana</ComboboxItem>
    <ComboboxEmpty>No results found.</ComboboxEmpty>
  </ComboboxContent>
</Combobox>
```
