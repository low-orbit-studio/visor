## Installation

```bash
npx visor add toast
```

## Usage

```tsx
// In your root layout:
import { Toaster } from '@/components/ui/toast/toast';

export default function Layout({ children }) {
  return (
    <>
      {children}
      <Toaster />
    </>
  );
}

// Anywhere in your app:
import { toast } from '@/components/ui/toast/toast';

toast('Event created');
```
