## Installation

```bash
npx visor add banner
```

## Usage

```tsx
import { Banner, BannerTitle, BannerDescription } from '@/components/ui/banner/banner';

<Banner intent="info">
  <BannerTitle>Heads up</BannerTitle>
  <BannerDescription>This is an informational message.</BannerDescription>
</Banner>
```
