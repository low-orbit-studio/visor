## Installation

```bash
npx visor add popover
```

## Usage

```tsx
import { Popover, PopoverTrigger, PopoverContent } from '@/components/ui/popover/popover';

<Popover>
  <PopoverTrigger>Open</PopoverTrigger>
  <PopoverContent>Place content here.</PopoverContent>
</Popover>
```
